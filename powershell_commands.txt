========================================
  COINBASE TRADER — POWERSHELL COMMANDS
========================================

 For troubleshooting specifically, the critical ones are:

  # 1. Is the backend alive?
  Invoke-RestMethod 'http://localhost:8001/api/cnn/status' | Select-Object device, cuda_available, training_active,
  train_status, scan_count

  # 2. Is training stuck / PID dead?
  Get-Content 'C:\Users\gl450\polymarket_app\backend\cnn_train_progress.json'

  # 3. Clear stale training lock
  $f = 'C:\Users\gl450\polymarket_app\backend\cnn_train_progress.json'
  $d = Get-Content $f | ConvertFrom-Json; $d.status = 'failed'; $d | ConvertTo-Json -Compress | Set-Content $f

  # 4. Recent errors in log
  Get-Content 'C:\Users\gl450\polymarket_app\backend\logs\backend.log' -Tail 100 | Select-String -Pattern
  "ERROR|WARNING"

  # 5. Are products tracked?
  Invoke-RestMethod 'http://localhost:8001/api/products?tracked=true&limit=5'

  # 6. Are CNN scans writing to DB?
  Invoke-RestMethod 'http://localhost:8001/api/cnn/scans?limit=3'

  # 7. GPU utilization
  nvidia-smi

# BUY signals only (filter client-side)

── TRAINING ─────────────────────────────────────────────────────────────────

# Check training progress file

# Trigger CNN training (POST)

# Check if training PID is still alive
$pid = (Get-Content 'C:\Users\gl450\polymarket_app\backend\cnn_train_progress.json' | ConvertFrom-Json).pid
Get-Process -Id $pid -ErrorAction SilentlyContinue | Select-Object Id, CPU, WorkingSet

# Clear stale training lock (if subprocess died but file says "running")
# (Edit the file manually or use the Write-Content below)
$f = 'C:\Users\gl450\polymarket_app\backend\cnn_train_progress.json'
$d = Get-Content $f | ConvertFrom-Json
$d.status = 'failed'
$d | ConvertTo-Json -Compress | Set-Content $f
